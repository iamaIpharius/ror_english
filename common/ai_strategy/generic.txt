	allowed = {
		NOT = {
			tag = ENG
			tag = USA
			tag = JAP
			tag = ITA
			tag = AST
		}
	}
	enable = {
		always = yes
	}
	ai_strategy = {
        type = invasion_unit_request
        country_trigger = { always = yes }
        value = -200
    }
}

    }

no_alliance_with_TUR = {
	allowed = {
		NOT = {
			tag = TUR
		}
	}
	enable = {
		has_global_flag = kis_firstbalkan_war
		TUR = {
			has_war_with = SER
		}
	}
	abort = {
		OR = {
			has_global_flag = kis_war
			any_country = {
				is_major = yes
				NOT = {
					tag = ITA
					tag = SER
				}
				has_war_with = TUR
			}
			has_global_flag = kis_firstbalkanwar_lose
			has_global_flag = kis_firstbalkanwar_victory
		}
	}	
	ai_strategy = {
		type = alliance
		id = "TUR"
		value = -200
	}
}
